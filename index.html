Result

 
import { Result, ok, err } from "@aicacia/core";

fetch("/api")
  .then((res) => ok(res))
  .catch((e) => err<Response>(e))
  .then((result) => {
    if (result.isOk()) {
      console.log(result.unwrap());
    } else {
      console.error(result.unwrapErr());
    }
  });
